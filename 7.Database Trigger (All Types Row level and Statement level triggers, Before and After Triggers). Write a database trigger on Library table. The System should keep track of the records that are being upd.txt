create database libary;

use libary;

CREATE TABLE library (
bno INT(5),  
bname VARCHAR(40),   
author VARCHAR(20),    
allowed_days INT(5) 
);

CREATE TABLE library_audit (
bno INT(5),
old_allowed_days INT(5),
new_allowed_days INT(5),
change_timestamp DATETIME
);

DELIMITER //
CREATE TRIGGER tr_update_library_audit
BEFORE UPDATE ON library
FOR EACH ROW
BEGIN
INSERT INTO library_audit (bno, old_allowed_days, new_allowed_days, change_timestamp)
VALUES (OLD.bno, OLD.allowed_days, NEW.allowed_days, NULL);
END //

DELIMITER ;
DELIMITER //

CREATE TRIGGER tr_delete_library_audit
BEFORE DELETE ON library
FOR EACH ROW
BEGIN
INSERT INTO library_audit (bno, old_allowed_days, new_allowed_days, change_timestamp)
VALUES (OLD.bno, OLD.allowed_days, NULL, NULL);
END //

DELIMITER ;
